# c(..., recursive = FALSE, use.names = TRUE)
x <- c(1:225)
class(x)
[1] "integer"
# matrix
# matrix(data, nrow, ncol, byrow, dimnames)
y <- matrix(1:225, nrow=15, ncol=15, byrow = FALSE)
